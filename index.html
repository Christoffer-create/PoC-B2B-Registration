async function fetchUserInfo() { try { const response = await fetch('/.auth/me'); if (response.ok) { const data = await response.json(); console.log('User Info:', data); // Debugging if (data.clientPrincipal) { const userInfo = data.clientPrincipal; console.log('Client Principal:', userInfo); // Debugging const email = userInfo.userDetails; const username = userInfo.userId; document.getElementById('user-info').innerHTML = `
User Information

Email: ${email}

Username: ${username}

`; } else { console.warn('User not authenticated, redirecting...'); window.location.href = "/.auth/login/aad"; // Redirect to login } } else { console.error('Error fetching user info:', response.statusText); } } catch (error) { console.error('Error fetching user info:', error); } } // Call the function to fetch user info fetchUserInfo();
